def area(a,b,c): # 3차원에서 세 점이 주어졌을 때 넓이
    m = (b[1]-a[1])*(c[2]-a[2]) - (b[2]-a[2])*(c[1]-a[1])
    n = (b[0]-a[0])*(c[2]-a[2]) - (b[2]-a[2])*(c[0]-a[0])
    p = (b[0]-a[0])*(c[1]-a[1]) - (b[1]-a[1])*(c[0]-a[0])
    aa = 0.5*(abs((m**2 + n**2 + p**2)**0.5))
    return aa
